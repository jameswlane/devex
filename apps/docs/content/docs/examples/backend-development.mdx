# Backend Development Environment

This example provides a comprehensive backend development environment supporting multiple programming languages, databases, container orchestration, and microservices development.

## Quick Start

<Steps>
  <Step>
    ### Download Configuration
    
    ```bash
    # Download the backend development configuration
    curl -fsSL https://raw.githubusercontent.com/jameswlane/devex/main/examples/backend-development.yaml > ~/.devex/backend-dev.yaml
    
    # Preview what will be installed
    devex config apply --file ~/.devex/backend-dev.yaml --dry-run
    ```
  </Step>
  
  <Step>
    ### Apply Configuration
    
    ```bash
    # Apply the configuration
    devex config apply --file ~/.devex/backend-dev.yaml
    
    # Install all tools
    devex install
    ```
  </Step>
  
  <Step>
    ### Verify Installation
    
    ```bash
    # Check installed tools
    node --version
    python3 --version
    go version
    docker --version
    kubectl version --client
    ```
  </Step>
</Steps>

    ```bash
    # Create Fastify project
    mkdir my-fastify-app
    cd my-fastify-app
    npm init -y
    
    # Install Fastify and plugins
    npm install fastify @fastify/cors @fastify/helmet

    # Create basic server
    cat > index.js << 'EOF'
    const fastify = require('fastify')({ logger: true })
    
    fastify.register(require('@fastify/cors'))
    fastify.register(require('@fastify/helmet'))
    fastify.register(require('@fastify/swagger'))
    
    fastify.get('/', async (request, reply) => {
      return { message: 'Hello World!' }
    })
    
    const start = async () => {
      try {
        await fastify.listen({ port: 3000 })
      } catch (err) {
        fastify.log.error(err)
        process.exit(1)
      }
    }
    
    start()
    EOF
    
    # Start server
    node index.js
    ```

  <Tab value="Koa.js">
    ```bash
    # Create Koa.js project
    mkdir my-koa-app
    cd my-koa-app
    npm init -y
    
    # Install Koa and middleware
    npm install koa @koa/cors @koa/router koa-helmet
    npm install koa-bodyparser koa-logger koa-jwt
    npm install --save-dev nodemon
    
    # Create basic app
    cat > app.js << 'EOF'
    const Koa = require('koa')
    const Router = require('@koa/router')
    const cors = require('@koa/cors')
    const helmet = require('koa-helmet')
    const bodyParser = require('koa-bodyparser')
    
    const app = new Koa()
    const router = new Router()
    
    app.use(helmet())
    app.use(cors())
    app.use(bodyParser())
    
    router.get('/', (ctx) => {
      ctx.body = { message: 'Hello World!' }
    })
    
    app.use(router.routes())
    app.use(router.allowedMethods())
    
    app.listen(3000, () => {
      console.log('Server running on port 3000')
    })
    EOF
    
    # Start development
    npx nodemon app.js
    ```

    ```bash
    # Create FastAPI project
    mkdir my-fastapi-app
    cd my-fastapi-app
    python -m venv venv
    source venv/bin/activate
    
    # Install FastAPI and dependencies
    pip install fastapi uvicorn[standard]
    pip install sqlalchemy alembic psycopg2-binary
    pip install pydantic[email] python-jose[cryptography]
    pip install pytest pytest-asyncio httpx
    
    # Create basic app structure
    mkdir -p app/{api,core,models,schemas,crud}
    
    # Create main.py
    cat > app/main.py << 'EOF'
    from fastapi import FastAPI
    from fastapi.middleware.cors import CORSMiddleware
    
    app = FastAPI(title="My API", version="1.0.0")
    
    app.add_middleware(
        CORSMiddleware,
        allow_origins=["*"],
        allow_credentials=True,
        allow_methods=["*"],
        allow_headers=["*"],
    )
    
    @app.get("/")
    async def root():
        return {"message": "Hello World"}
    
    @app.get("/health")
    async def health_check():
        return {"status": "healthy"}
    EOF
    
    # Start development server
    uvicorn app.main:app --reload
    ```

    ```bash
    # Create Flask project
    mkdir my-flask-app
    cd my-flask-app
    python -m venv venv
    source venv/bin/activate
    
    # Install Flask and extensions
    pip install flask flask-sqlalchemy flask-migrate
    pip install flask-jwt-extended flask-cors
    pip install marshmallow flask-marshmallow
    pip install pytest pytest-flask
    
    # Create app structure
    mkdir -p app/{api,models,schemas}
    
    # Create main app
    cat > app/__init__.py << 'EOF'
    from flask import Flask
    from flask_sqlalchemy import SQLAlchemy
    from flask_migrate import Migrate
    from flask_cors import CORS
    
    db = SQLAlchemy()
    migrate = Migrate()
    
    def create_app():
        app = Flask(__name__)
        app.config['SQLALCHEMY_DATABASE_URI'] = 'sqlite:///app.db'
        app.config['SECRET_KEY'] = 'dev-secret-key'
        
        db.init_app(app)
        migrate.init_app(app, db)
        CORS(app)
        
        from app.api import bp as api_bp
        app.register_blueprint(api_bp, url_prefix='/api')
        
        return app
    ```
