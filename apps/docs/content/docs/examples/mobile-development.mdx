# Mobile Development Environment

This example provides a comprehensive mobile development environment supporting React Native, Flutter, native iOS, and Android development with proper tooling, emulators, and testing capabilities.

## Quick Start

<Steps>
  <Step>
    ### Download Configuration
    
    ```bash
    # Download the mobile development configuration
    curl -fsSL https://raw.githubusercontent.com/jameswlane/devex/main/examples/mobile-development.yaml > ~/.devex/mobile-dev.yaml
    
    # Preview what will be installed
    devex config apply --file ~/.devex/mobile-dev.yaml --dry-run
    ```
  </Step>
  
  <Step>
    ### Apply Configuration
    
    ```bash
    # Apply the configuration
    devex config apply --file ~/.devex/mobile-dev.yaml
    
    # Install all tools
    devex install
    ```
  </Step>
  
  <Step>
    ### Verify Installation
    
    ```bash
    # Check installed tools
    node --version
    react-native --version
    flutter --version  # if Flutter was selected
    adb version
    ```
  </Step>
</Steps>

    # Create bare React Native project
    npx react-native@latest init MyBareApp --template react-native-template-typescript
    cd MyBareApp

    npm install react-native-permissions
    npm install react-native-keychain
    npm install react-native-biometrics
    npm install react-native-device-info
    
    # Install development tools
    npm install --save-dev reactotron-react-native
    npm install --save-dev flipper-react-native
    
    # Configure Metro for better performance
    cat > metro.config.js << 'EOF'
    const { getDefaultConfig } = require('metro-config');
    
    module.exports = (async () => {
      const {
        resolver: { sourceExts, assetExts },
      } = await getDefaultConfig();
      
      return {
        transformer: {
          babelTransformerPath: require.resolve('react-native-svg-transformer'),
        },
        resolver: {
          assetExts: assetExts.filter(ext => ext !== 'svg'),
          sourceExts: [...sourceExts, 'svg'],
        },
      };
    })();
    EOF
    
    # Setup linking for iOS
    cd ios && pod install && cd ..
    ```

#!/bin/bash
# scripts/mobile-dev-setup.sh
# Mobile development environment automation

set -e

PLATFORM="${1:-all}"  # react-native, flutter, android, ios, all

echo "📱 Setting up mobile development environment for: $PLATFORM"

setup_react_native() {
    echo "⚛️ Setting up React Native environment..."
    
    # Install React Native CLI
    npm install -g @react-native-community/cli
    npm install -g @expo/cli
    
    # Install development tools
    npm install -g react-devtools
    npm install -g flipper-pkg
    npm install -g reactotron-cli
    
    echo "✅ React Native setup complete"
}

setup_flutter() {
    echo "🐦 Setting up Flutter environment..."
    
    # Verify Flutter installation
    if command -v flutter &> /dev/null; then
        flutter doctor
        flutter config --enable-web
        flutter config --enable-linux-desktop
        flutter config --enable-macos-desktop
        flutter config --enable-windows-desktop
    else
        echo "❌ Flutter not found. Please install Flutter first."
        exit 1
    fi
    
    echo "✅ Flutter setup complete"
}

setup_android() {
    echo "🤖 Setting up Android development..."
    
    # Set Android environment variables
    export ANDROID_HOME="$HOME/Android/Sdk"
    export PATH="$PATH:$ANDROID_HOME/emulator"
    export PATH="$PATH:$ANDROID_HOME/tools"
    export PATH="$PATH:$ANDROID_HOME/tools/bin"
    export PATH="$PATH:$ANDROID_HOME/platform-tools"
    
    # Create AVD if it doesn't exist
    if ! avdmanager list avd | grep -q "DevEx_Pixel_6_API_33"; then
        echo "📱 Creating Android Virtual Device..."
        avdmanager create avd -n "DevEx_Pixel_6_API_33" \
            -k "system-images;android-33;google_apis;x86_64" \
            --device "pixel_6"
    fi
    
    echo "✅ Android setup complete"
}

setup_ios() {
    if [[ "$OSTYPE" == "darwin"* ]]; then
        echo "🍎 Setting up iOS development..."
        
        # Install CocoaPods
        if ! command -v pod &> /dev/null; then
            sudo gem install cocoapods
        fi
        
        # Setup pod repo
        pod setup
        
        echo "✅ iOS setup complete"
    else
        echo "ℹ️ iOS development only available on macOS"
    fi
}

case $PLATFORM in
    "react-native")
        setup_react_native
        setup_android
        setup_ios
        ;;
    "flutter")
        setup_flutter
        setup_android
        setup_ios
        ;;
    "android")
        setup_android
        ;;
    "ios")
        setup_ios
        ;;
    "all")
        setup_react_native
        setup_flutter
        setup_android
        setup_ios
        ;;
    *)
        echo "❌ Unknown platform: $PLATFORM"
        echo "Usage: $0 [react-native|flutter|android|ios|all]"
        exit 1
        ;;
esac

echo "🎉 Mobile development environment setup complete for $PLATFORM!"
```

### Build and Deployment Automation

```bash
#!/bin/bash
# scripts/mobile-build.sh
# Automated mobile app building and deployment

set -e

PROJECT_TYPE="${1:-react-native}"  # react-native, flutter, android, ios
BUILD_TYPE="${2:-debug}"           # debug, release
PLATFORM="${3:-both}"              # android, ios, both

echo "🏗️ Building $PROJECT_TYPE app ($BUILD_TYPE) for $PLATFORM"

build_react_native() {
    echo "⚛️ Building React Native app..."
    
    if [[ "$PLATFORM" == "android" || "$PLATFORM" == "both" ]]; then
        echo "🤖 Building Android..."
        if [[ "$BUILD_TYPE" == "release" ]]; then
            cd android && ./gradlew assembleRelease && cd ..
        else
            cd android && ./gradlew assembleDebug && cd ..
        fi
    fi
    
    if [[ "$PLATFORM" == "ios" || "$PLATFORM" == "both" ]]; then
        if [[ "$OSTYPE" == "darwin"* ]]; then
            echo "🍎 Building iOS..."
            cd ios && pod install && cd ..
            if [[ "$BUILD_TYPE" == "release" ]]; then
                npx react-native run-ios --configuration Release
            else
                npx react-native run-ios
            fi
        else
            echo "ℹ️ iOS build skipped (not on macOS)"
        fi
    fi
}

build_flutter() {
    echo "🐦 Building Flutter app..."
    
    if [[ "$PLATFORM" == "android" || "$PLATFORM" == "both" ]]; then
        echo "🤖 Building Android..."
        if [[ "$BUILD_TYPE" == "release" ]]; then
            flutter build apk --release
            flutter build appbundle --release
        else
            flutter build apk --debug
        fi
    fi
    
    if [[ "$PLATFORM" == "ios" || "$PLATFORM" == "both" ]]; then
        if [[ "$OSTYPE" == "darwin"* ]]; then
            echo "🍎 Building iOS..."
            if [[ "$BUILD_TYPE" == "release" ]]; then
                flutter build ios --release
            else
                flutter build ios --debug
            fi
        else
            echo "ℹ️ iOS build skipped (not on macOS)"
        fi
    fi
}

case $PROJECT_TYPE in
    "react-native")
        build_react_native
        ;;
    "flutter")
        build_flutter
        ;;
    *)
        echo "❌ Unknown project type: $PROJECT_TYPE"
        exit 1
        ;;
esac

echo "✅ Build complete!"
```
