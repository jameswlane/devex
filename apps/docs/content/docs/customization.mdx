# Customization

DevEx is designed to be highly customizable, allowing you to tailor every aspect of your development environment to your specific needs and preferences.

## Customization Philosophy

DevEx follows a "configuration over convention" approach while maintaining sensible defaults:

- **Defaults First**: Start with proven defaults that work for most developers
- **Override Selectively**: Customize only what you need to change
- **Team Consistency**: Share configurations across teams and projects
- **Platform Awareness**: Adapt automatically to different operating systems

## Configuration Hierarchy

Understanding the configuration hierarchy helps you customize effectively:

```
1. Environment Variables (highest priority)
2. Command-line flags
3. User overrides (~/.devex/)
4. Default configurations (~/.local/share/devex/config/)
5. Built-in fallbacks (lowest priority)
```

## Creating Custom Application Definitions

### Basic Custom Application

Create `~/.devex/applications.yaml` to add or override applications:

```yaml
categories:
  development:
    # Add a new application
    neovim:
      description: "Modern Vim-based text editor"
      linux:
        apt: "neovim"
        dnf: "neovim"
        pacman: "neovim"
        flatpak: "io.neovim.nvim"
      mac:
        brew: "neovim"
      windows:
        winget: "Neovim.Neovim"
        chocolatey: "neovim"
        scoop: "neovim"
      verify:
        command: "nvim --version"
      post_install:
        - "mkdir -p ~/.config/nvim"
        - "echo 'set number' > ~/.config/nvim/init.vim"
```

### Application with Complex Installation

```yaml
categories:
  development:
    jetbrains-toolbox:
      description: "JetBrains IDEs management tool"
      linux:
        method: "jetbrains_toolbox"  # Custom installer
      mac:
        method: "jetbrains_toolbox"
      windows:
        method: "jetbrains_toolbox"
      verify:
        command: "jetbrains-toolbox --version"
        alternative: "ls ~/.local/share/JetBrains/Toolbox"
      dependencies:
        - "curl"
        - "tar"
      post_install:
        - "jetbrains-toolbox --install-desktop-entry"
      config:
        install_path: "~/.local/share/JetBrains/Toolbox"
        auto_update: true
```

Create a custom installer for complex applications:

    "context"
    "fmt"
    "os/exec"

    dryRun bool
    logger types.Logger
}

func NewJetBrainsInstaller(dryRun bool, logger types.Logger) *JetBrainsInstaller {
    return &JetBrainsInstaller{
        dryRun: dryRun,
        logger: logger,
    }
}

func (j *JetBrainsInstaller) Install(ctx context.Context, app types.Application) error {
    if j.dryRun {
        j.logger.Info("Would install JetBrains Toolbox")
        return nil
    }
    
    // Download and install JetBrains Toolbox
    downloadURL := "https://download.jetbrains.com/toolbox/jetbrains-toolbox.tar.gz"
    
    cmd := exec.CommandContext(ctx, "curl", "-L", downloadURL, "-o", "/tmp/jetbrains-toolbox.tar.gz")
    if err := cmd.Run(); err != nil {
        return fmt.Errorf("failed to download JetBrains Toolbox: %w", err)
    }
    
    // Extract and install
    cmd = exec.CommandContext(ctx, "tar", "-xzf", "/tmp/jetbrains-toolbox.tar.gz", "-C", "/tmp")
    if err := cmd.Run(); err != nil {
        return fmt.Errorf("failed to extract JetBrains Toolbox: %w", err)
    }

    return nil
}

    cmd := exec.CommandContext(ctx, "jetbrains-toolbox", "--version")
    return cmd.Run() == nil, nil
}

## Project-Specific Customization

### Project Configuration Files

Create `.devex/` directory in your project root:

```bash
my-project/
├── .devex/
│   ├── applications.yaml
│   ├── environment.yaml
│   └── system.yaml
├── src/
└── package.json
```

### Project Applications

```yaml
# my-project/.devex/applications.yaml
categories:
  development:
    # Project-specific tools
    terraform: true
    aws_cli: true
    kubectl: true
    
  languages:
    nodejs:
      version: "18.17.0"  # Specific version for this project
      packages:
        - "typescript@5.0.0"
        - "@nestjs/cli@10.0.0"
```

### Load Project Configuration

```bash
cd my-project

# Install project-specific environment
devex install --config .devex/

# Or set environment variable
export DEVEX_CONFIG_DIR="$(pwd)/.devex"
devex install
```

## Team Customization

### Shared Team Configuration

Create a team configuration repository:

```bash
# Create team config repository
mkdir devex-team-config
cd devex-team-config

# Team applications
cat > applications.yaml << 'EOF'
categories:
  development:
    git: true
    vscode: true
    docker: true
    kubernetes_tools: true
    
  languages:
    nodejs:
      version: "18.17.0"
    python:
      version: "3.11.4"
    go:
      version: "1.20.6"
      
  databases:
    postgresql: true
    redis: true
EOF

# Team environment settings
cat > environment.yaml << 'EOF'
languages:
  nodejs:
    global_packages:
      - "yarn@1.22.19"
      - "typescript@5.0.0"
      - "@nestjs/cli@10.0.0"
      
shell:
  oh_my_zsh:
    theme: "robbyrussell"
    plugins:
      - "git"
      - "docker"
      - "kubectl"
      - "node"
EOF

# Version control
git init
git add .
git commit -m "Initial team DevEx configuration"
git remote add origin https://github.com/yourorg/devex-team-config.git
git push -u origin main
```

### Team Member Setup

```bash
# Clone team configuration
git clone https://github.com/yourorg/devex-team-config.git ~/.devex

# Install team development environment
devex install
devex system

# Stay updated with team changes
cd ~/.devex
git pull
devex install
```

## Advanced Customization Techniques

### Conditional Configuration

```yaml
# Platform-specific configuration
categories:
  development:
    terminal:
      linux:
        apt: "gnome-terminal"
        dnf: "gnome-terminal"
      mac:
        brew: "iterm2 --cask"
      windows:
        winget: "Microsoft.WindowsTerminal"
        
    browser:
      condition: "desktop_environment != 'server'"
      linux:
        flatpak: "org.mozilla.firefox"
      mac:
        brew: "firefox --cask"
      windows:
        winget: "Mozilla.Firefox"
```

### Environment-Based Configuration

```yaml
# Different configs for different environments
profiles:
  development:
    categories:
      development: true
      languages: true
      databases: true
      optional: true
      
  production:
    categories:
      system: true
      databases: true
      development:
        git: true
        docker: true
        
  ci:
    categories:
      development:
        git: true
        docker: true
      languages:
        nodejs: true
        python: true
```

Load specific profile:

```bash
devex install --profile development
devex install --profile production
```

### Custom Validation Rules

```yaml
# Custom validation for applications
categories:
  development:
    docker:
      validation:
        pre_install:
          - command: "which systemctl"
            error: "systemd is required for Docker installation"
        post_install:
          - command: "docker info"
            timeout: 30
            retry: 3
          - command: "docker run hello-world"
            cleanup: "docker rmi hello-world"
```
